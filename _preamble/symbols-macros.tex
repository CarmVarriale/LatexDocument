% Additional shortcuts for subscripts and superscripts (also nested)
\newcommand*{\gsub}[2]{\ensuremath{\gls{#1}[_{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gsubrm}[2]{\ensuremath{\gls{#1}[_{%
					\ifglsentryexists{#2}{%
						\mathrm{\gls{#2}}}{%
						\mathrm{#2}}}]}}
\newcommand*{\gbarsub}[2]{\ensuremath{\glsbar{#1}[_{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gbarsubrm}[2]{\ensuremath{\glsbar{#1}[_{%
					\ifglsentryexists{#2}{\mathrm{\gls{#2}}}{\mathrm{#2}}}]}}
\newcommand*{\gbarsup}[2]{\ensuremath{\glsbar{#1}[^{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gbarsuprm}[2]{\ensuremath{\glsbar{#1}[^{%
					\ifglsentryexists{#2}{\mathrm{\gls{#2}}}{\mathrm{#2}}}]}}
\newcommand*{\gdotsub}[2]{\ensuremath{\glsdot{#1}[_{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gdotsubrm}[2]{\ensuremath{\glsdot{#1}[_{%
					\ifglsentryexists{#2}{\mathrm{\gls{#2}}}{\mathrm{#2}}}]}}
\newcommand*{\gboldsub}[2]{\ensuremath{\glsbold{#1}[_{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gsubbold}[2]{\ensuremath{\gls{#1}[_{%
					\ifglsentryexists{#2}{\glsbold{#2}}{\bm{#2}}}]}}
\newcommand*{\gboldsubbold}[2]{\ensuremath{\glsbold{#1}[_{%
					\ifglsentryexists{#2}{\glsbold{#2}}{#2}}]}}
\newcommand*{\gboldsubrm}[2]{\ensuremath{\glsbold{#1}[_{%
					\ifglsentryexists{#2}{\mathrm{\gls{#2}}}{\mathrm{#2}}}]}}


\newcommand*{\gsup}[2]{\ensuremath{\gls{#1}[^{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gboldsup}[2]{\ensuremath{\glsbold{#1}[^{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}
\newcommand*{\gsuprm}[2]{\ensuremath{\gls{#1}[^{%
					\ifglsentryexists{#2}{\mathrm{\gls{#2}}}{\mathrm{#2}}}]}}
\newcommand*{\gdotsup}[2]{\ensuremath{\glsdot{#1}[^{%
					\ifglsentryexists{#2}{\gls{#2}}{#2}}]}}


\newcommand*{\gsubsub}[3]{\ensuremath{\gls{#1}[_{%
					\ifglsentryexists{#2}{%
						\gls{#2}[_{\ifglsentryexists{#3}{\gls{#3}}{#3}}]}{%
						#2_{\ifglsentryexists{#3}{\gls{#3}}{#3}}}}]}}

\newcommand*{\gsubsubsub}[4]{\ensuremath{%
\gls{#1}[_{%
\ifglsentryexists{#2}%
{\gls{#2}[_{\ifglsentryexists{#3}%
			{\gls{#3}[_{\ifglsentryexists{#4}{\gls{#4}}{#4}}]}%
			{{#3}_{\ifglsentryexists{#4}{\gls{#4}}{#4}}}%
		}]%
}%
{{#2}_{\ifglsentryexists{#3}%
{\gls{#3}[_{\ifglsentryexists{#4}{\gls{#4}}{#4}}]}%
{{#3}_{\ifglsentryexists{#4}{\gls{#4}}{#4}}}%
}}}]}}%
